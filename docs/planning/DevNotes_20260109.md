```javascript
function handleSignup(e) {
  e.preventDefault();

  const nickname = document.getElementById('signup-id').value.trim();
  const password = document.getElementById('signup-password').value.trim();
  const passwordConfirm = document.getElementById('signup-password-confirm').value.trim();

  if (password !== passwordConfirm) {
    showAlert('비밀번호가 일치하지 않습니다.', 'warning');
    return;
  }

  showLoading(true);
  google.script.run
    .withSuccessHandler(function(result) {
      showLoading(false);
      if (result.success) {
        showAlert(result.message, 'success');
        setTimeout(() => showLoginScreen(), 2000);
      }
    })
    .registerParticipant(nickname, password);
}
```

## 2. 로그인 로직 수정

### 2.1 백엔드 함수 (Code.gs)

**함수명:** `loginParticipant(nickname, password)`
**위치:** Code.gs 라인 204-310

**기존 로직 (수정 전):**
- 계정 없으면 자동 생성
- 계정 있으면 최근접속 업데이트

**새 로직 (수정 후):**

```javascript
function loginParticipant(nickname, password) {
  // 1. 계정 찾기
  for (let i = 1; i < data.length; i++) {
    const rowNickname = String(data[i][0] || '').trim();

    if (rowNickname === nickname.trim()) {
      // 2. 비밀번호 확인
      const rowPassword = String(data[i][1] || '').trim();
      if (rowPassword !== password.trim()) {
        return { success: false, message: '비밀번호가 일치하지 않습니다.' };
      }

      // 3. 승인여부 확인
      const approvalStatus = String(data[i][2] || '').trim();

      if (approvalStatus === '가입대기') {
        return {
          success: false,
          status: 'pending',
          message: '관리자의 승인을 기다리고 있습니다. 승인 후 로그인이 가능합니다.'
        };
      }

      if (approvalStatus === '승인거부') {
        return {
          success: false,
          status: 'rejected',
          message: '가입이 거부되었습니다. 관리자에게 문의해주세요.'
        };
      }

      if (approvalStatus === '말소') {
        return {
          success: false,
          status: 'deleted',
          message: '말소된 계정입니다. 관리자에게 문의해주세요.'
        };
      }

      // 4. '정상' 상태 - 로그인 허용
      if (approvalStatus === '정상') {
        // 최근접속 업데이트
        const lastAccess = data[i][4] ?
          Utilities.formatDate(new Date(data[i][4]), 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss') :
          '처음 접속';
        sheet.getRange(i + 1, 5).setValue(now);

        return {
          success: true,
          isFirstTime: !data[i][4],
          lastAccess: lastAccess,
          nickname: nickname,
          message: '로그인 성공!'
        };
      }

      // 5. 알 수 없는 상태
      return {
        success: false,
        message: '계정 상태를 확인할 수 없습니다. 관리자에게 문의해주세요.'
      };
    }
  }

  // 6. 계정을 찾지 못함
  return {
    success: false,
    message: '존재하지 않는 계정입니다. 회원가입을 먼저 진행해주세요.'
  };
}
```

**변경 사항:**
- ❌ 자동 계정 생성 제거
- ✅ 승인여부 체크 추가
- ✅ 상태별 오류 메시지 반환
- ✅ '정상' 상태만 로그인 허용
- ✅ 계정 없으면 회원가입 안내
