## 3. 날짜 포맷 함수 분리

### 3.1 문제점

- `formatDate()` 함수가 여러 함수 내부에 인라인으로 중복 정의되어 있었음
- 동일한 로직이 5곳 이상에서 반복

### 3.2 해결

**전역 함수로 분리:**

```javascript
/**
 * 날짜 값을 'yyyy-MM-dd HH:mm:ss' 형식의 문자열로 변환
 * Date 객체, 문자열, 또는 빈 값을 처리합니다.
 *
 * @param {Date|string|null} dateValue - 변환할 날짜 값
 * @returns {string} 포맷된 날짜 문자열 또는 빈 문자열
 */
function formatDate(dateValue) {
  if (!dateValue) return '';

  if (dateValue instanceof Date) {
    return Utilities.formatDate(dateValue, 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
  }

  const dateStr = String(dateValue);

  // 이미 올바른 형식이면 그대로 반환
  if (/^\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}$/.test(dateStr)) {
    return dateStr;
  }

  // 다른 형식이면 파싱 후 변환 시도
  try {
    return Utilities.formatDate(new Date(dateValue), 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
  } catch (e) {
    return dateStr;
  }
}
```

### 3.3 장점

- 코드 중복 제거
- 날짜 포맷 로직 일관성 확보
- 유지보수 용이

## 8. 패키지 재개시 종료일 직접 입력

### 8.1 문제점

- 기존에는 패키지 재개 시 무조건 7일 후로 종료일이 설정됨
- 관리자가 원하는 종료일을 직접 지정할 수 없음

**기존 코드:**
```javascript
function resumePackage(password) {
  const now = new Date();
  const newEndDate = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000);  // 7일 고정
  const endDateFormatted = Utilities.formatDate(newEndDate, 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
  return reopenSurvey(password, endDateFormatted);
}
```

### 8.2 해결 - Code.gs

**수정된 함수:**
```javascript
/**
 * 패키지 재개 (사용자가 종료일 직접 지정)
 * @param {string} password - 패키지 비밀번호(코드)
 * @param {string} newEndDate - 새 종료일 (yyyy-MM-dd HH:mm:ss 형식, 미지정시 7일 후)
 */
function resumePackage(password, newEndDate) {
  // newEndDate가 제공되지 않으면 기본 7일 후로 설정
  if (!newEndDate) {
    const now = new Date();
    const defaultEndDate = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000);
    newEndDate = Utilities.formatDate(defaultEndDate, 'Asia/Seoul', 'yyyy-MM-dd HH:mm:ss');
  }
  return reopenSurvey(password, newEndDate);
}
```

### 8.3 해결 - index.html

**수정된 함수:**
```javascript
function resumePackage(password) {
  // 기본값: 7일 후
  const defaultDate = new Date();
  defaultDate.setDate(defaultDate.getDate() + 7);
  const year = defaultDate.getFullYear();
  const month = String(defaultDate.getMonth() + 1).padStart(2, '0');
  const day = String(defaultDate.getDate()).padStart(2, '0');
  const hours = String(defaultDate.getHours()).padStart(2, '0');
  const minutes = String(defaultDate.getMinutes()).padStart(2, '0');
  const defaultDateStr = `${year}-${month}-${day} ${hours}:${minutes}:00`;

  // 종료일 입력 프롬프트
  const newEndDate = prompt(
    '새 종료일시를 입력하세요.\n\n형식: YYYY-MM-DD HH:MM:SS\n예시: 2026-02-15 23:59:00\n\n(취소하려면 취소 버튼을 누르세요)',
    defaultDateStr
  );

  if (!newEndDate) return;  // 취소

  // 날짜 형식 검증
  const datePattern = /^\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}$/;
  if (!datePattern.test(newEndDate)) {
    alert('❌ 날짜 형식이 올바르지 않습니다.\n\n올바른 형식: YYYY-MM-DD HH:MM:SS\n예시: 2026-02-15 23:59:00');
    return;
  }

  // 미래 날짜인지 검증
  const inputDate = new Date(newEndDate.replace(' ', 'T'));
  if (inputDate <= new Date()) {
    alert('❌ 종료일은 현재 시간보다 미래여야 합니다.');
    return;
  }

  if (!confirm(`종료일: ${newEndDate}\n\n이 설정으로 패키지를 재개하시겠습니까?`)) return;

  showLoading();
  google.script.run
    .withSuccessHandler(function(result) {
      showLoading(false);
      if (result.success) {
        alert('✅ 패키지가 재개되었습니다.\n종료일: ' + newEndDate);
        loadCompletedPackages();
      } else {
        alert('❌ ' + result.message);
      }
    })
    .withFailureHandler(function(err) {
      showLoading(false);
      alert('오류: ' + err.message);
    })
    .resumePackage(password, newEndDate);
}
```

## 9. 종료일 지난 패키지 자동 완료 처리

### 9.1 문제점

- 종료일이 지난 패키지가 여전히 D(활성) 상태로 남아있음
- 피검자용 웹앱에서 이미 종료된 설문이 여전히 표시됨
- 예: PKG000078 (종료일: 2026-01-30) → 2026-02-03에도 D 상태

### 9.2 해결 - 시간 기반 트리거

**GAS의 Time-driven Trigger를 활용한 자동 완료 처리:**

```javascript
// =================================================================================
// 10. AUTO-COMPLETE TRIGGER (자동 완료 처리)
// =================================================================================

/**
 * 종료일이 지난 활성 패키지를 자동으로 완료 처리
 * 시간 기반 트리거로 주기적 실행 (예: 매시간)
 */
function autoCompleteExpiredPackages() {
  try {
    const ss = SpreadsheetApp.openById(getConfigValue('PACKAGE_LIST_SHEET_ID'));
    const sheet = ss.getSheetByName('패키지 목록');

    if (!sheet) {
      Logger.log('패키지 목록 시트를 찾을 수 없습니다.');
      return;
    }

    const data = sheet.getDataRange().getValues();
    const now = new Date();
    let completedCount = 0;

    // 패키지 목록: 10컬럼 구조 (H열=종료일, J열=상태)
    for (let i = 1; i < data.length; i++) {
      const status = String(data[i][9] || '').trim().toUpperCase();
      const endDate = data[i][7];  // H열: 종료일

      // D(활성) 상태이고 종료일이 있는 경우
      if (status === STATUS.ACTIVE && endDate) {
        const endDateTime = new Date(endDate);

        // 종료일이 현재 시간보다 이전이면 완료 처리
        if (endDateTime < now) {
          const endDateFormatted = formatDate(now);
          sheet.getRange(i + 1, 8).setValue(endDateFormatted);  // H열: 실제 종료일 업데이트
          sheet.getRange(i + 1, 10).setValue(STATUS.COMPLETED); // J열: 상태를 C로 변경
          completedCount++;

          Logger.log(`패키지 자동 완료: ${data[i][0]} (종료일: ${endDate})`);
        }
      }
    }

    if (completedCount > 0) {
      Logger.log(`자동 완료 처리 완료: ${completedCount}개 패키지`);
    }

  } catch (error) {
    Logger.log('자동 완료 처리 오류: ' + error.message);
  }
}

/**
 * 자동 완료 트리거 설치 (매시간 실행)
 */
function installAutoCompleteTrigger() {
  // 기존 동일 트리거 삭제 (중복 방지)
  const triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(trigger => {
    if (trigger.getHandlerFunction() === 'autoCompleteExpiredPackages') {
      ScriptApp.deleteTrigger(trigger);
      Logger.log('기존 자동 완료 트리거 삭제됨');
    }
  });

  // 새 트리거 설치 (매시간)
  ScriptApp.newTrigger('autoCompleteExpiredPackages')
    .timeBased()
    .everyHours(1)
    .create();

  Logger.log('자동 완료 트리거 설치 완료 (매시간 실행)');
}

/**
 * 자동 완료 트리거 제거
 */
function uninstallAutoCompleteTrigger() {
  const triggers = ScriptApp.getProjectTriggers();
  let removed = 0;

  triggers.forEach(trigger => {
    if (trigger.getHandlerFunction() === 'autoCompleteExpiredPackages') {
      ScriptApp.deleteTrigger(trigger);
      removed++;
    }
  });

  Logger.log(`자동 완료 트리거 ${removed}개 제거됨`);
}

/**
 * 수동으로 만료된 패키지 일괄 완료 처리
 */
function manualCompleteExpiredPackages() {
  autoCompleteExpiredPackages();
  return { success: true, message: '만료된 패키지 완료 처리가 실행되었습니다.' };
}
```
